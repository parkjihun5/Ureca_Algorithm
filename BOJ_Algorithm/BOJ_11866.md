### BOJ 11866번 풀이
https://www.acmicpc.net/problem/11866

```
import java.util.Scanner;
import java.util.LinkedList;
import java.util.Queue;

public class Main {
    public static void main(String[] args) {
        // 1. 입력을 받기 위한 준비
        Scanner sc = new Scanner(System.in);
        
        // 2. 출력을 모아두기 위한 StringBuilder (결과를 <3, 6, 2...> 형태로 만들기 위함)
        StringBuilder sb = new StringBuilder();
        
        // 3. N(인원수)과 K(제거 순서) 입력 받기
        int N = sc.nextInt(); 
        int K = sc.nextInt(); 
        
        // 4. Queue 인터페이스를 LinkedList 구현체로 생성
        // LinkedList는 앞뒤 데이터 삽입/삭제가 빨라 큐 구현에 적합함
        Queue<Integer> queue = new LinkedList<>();
        
        // 5. 1부터 N까지의 숫자를 큐에 순서대로 담기
        // 예: N=7이면 [1, 2, 3, 4, 5, 6, 7]
        for (int i = 1; i <= N; i++) {
            queue.add(i);
        }
        
        // 6. 결과 문자열의 시작인 '<' 추가
        sb.append("<");
        
        // 7. 큐가 빌 때까지(모든 사람이 제거될 때까지) 반복 실행
        while (!queue.isEmpty()) {
            
            // [핵심 로직] K번째 사람을 찾기 위해 K-1번 동안 맨 앞사람을 뒤로 보냄
            // 예: K=3이면, 2번은 뒤로 보내고 3번째 사람을 추출함
            for (int i = 0; i < K - 1; i++) {
                // 맨 앞의 사람을 꺼내서(poll)
                int temp = queue.poll();
                // 다시 맨 뒤로 넣음(add) -> 원형으로 도는 효과
                queue.add(temp);
            }
            
            // 8. 드디어 K번째에 도달한 사람을 큐에서 완전히 꺼냄
            int removedPerson = queue.poll();
            sb.append(removedPerson);
            
            // 9. 큐에 아직 사람이 남아있다면 숫자 뒤에 쉼표와 공백 추가
            // 마지막 숫자 뒤에는 쉼표가 붙지 않게 하기 위한 조건문
            if (!queue.isEmpty()) {
                sb.append(", ");
            }
        }
        
        // 10. 결과 문자열의 끝인 '>' 추가
        sb.append(">");
        
        // 11. 최종적으로 완성된 요세푸스 순열 출력
        System.out.println(sb.toString());
        
        // 12. 스캐너 자원 닫기
        sc.close();
    }
}
```
