### BOJ 1000번 풀이
https://www.acmicpc.net/problem/1000

```
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        
        int A = sc.nextInt();
        int B = sc.nextInt();
        
        int C = A + B;
        
        System.out.print(C);

        sc.close();
    
    }
}
```
