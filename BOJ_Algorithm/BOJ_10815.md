### BOJ 10815번 풀이
https://www.acmicpc.net/problem/10815

Scanner 사용 ver.
```
import java.util.*;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        Set<Integer> set = new HashSet<>();
        
        int N = sc.nextInt(); // 입력받을 데이터 개수
        for (int i = 0; i < N; i++) {
            set.add(sc.nextInt());
        }
        
        int M = sc.nextInt();
        
        int[] arr = new int[M];
        
        for (int i = 0; i < M; i++) {
            arr[i] = sc.nextInt();
        }
        
        for (int x : arr) {
            if (set.contains(x)) {
                System.out.print(1 + " ");
            } else {
                System.out.print(0 + " ");
            }
        }
        
    }
}
```

BufferedReader 사용 ver
```
import java.util.*;
import java.io.*; // BufferedReader와 InputStreamReader를 위해 추가

public class Main {
    public static void main(String[] args) throws Exception { // 예외 처리를 위해 필수
        // 1. 빠른 입력을 위한 조합
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        // 2. 빠른 출력을 위한 창고
        StringBuilder sb = new StringBuilder();
        
        // N 읽기
        int N = Integer.parseInt(br.readLine());
        Set<Integer> set = new HashSet<>();
        
        // 한 줄에 들어온 N개의 숫자를 쪼개서 Set에 넣기
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < N; i++) {
            set.add(Integer.parseInt(st.nextToken()));
        }
        
        // M 읽기
        int M = Integer.parseInt(br.readLine());
        
        // 한 줄에 들어온 M개의 숫자를 쪼개서 확인하기
        st = new StringTokenizer(br.readLine());
        for (int i = 0; i < M; i++) {
            int x = Integer.parseInt(st.nextToken());
            
            // 바로 출력하지 않고 sb에 모으기
            if (set.contains(x)) {
                sb.append(1).append(" ");
            } else {
                sb.append(0).append(" ");
            }
        }
        
        // 3. 마지막에 단 한 번만 출력!
        System.out.println(sb.toString());
    }
}
```
