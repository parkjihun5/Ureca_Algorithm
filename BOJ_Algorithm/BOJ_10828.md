### Scanner와 deque를 활용한 ver
```
import java.util.ArrayDeque;
import java.util.Deque;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // 결과를 한 번에 출력하기 위해 StringBuilder 사용 (출력 최적화)
        StringBuilder sb = new StringBuilder();
        
        // 스택으로 사용할 ArrayDeque 선언
        Deque<Integer> stack = new ArrayDeque<>();
        
        int N = sc.nextInt(); // 명령의 수
        
        for (int i = 0; i < N; i++) {
            String command = sc.next();
            
            switch (command) {
                case "push":
                    int num = sc.nextInt();
                    stack.push(num); // LIFO 구조로 맨 앞에 넣기
                    break;
                    
                case "pop":
                    if (stack.isEmpty()) {
                        sb.append("-1\n");
                    } else {
                        sb.append(stack.pop()).append("\n"); // 맨 앞의 요소를 빼고 출력
                    }
                    break;
                    
                case "size":
                    sb.append(stack.size()).append("\n");
                    break;
                    
                case "empty":
                    sb.append(stack.isEmpty() ? "1\n" : "0\n");
                    break;
                    
                case "top":
                    if (stack.isEmpty()) {
                        sb.append("-1\n");
                    } else {
                        sb.append(stack.peek()).append("\n"); // 빼지 않고 확인만 하기
                    }
                    break;
            }
        }
        
        // 결과 출력
        System.out.print(sb.toString());
        sc.close();
    }
}

### BufferedReader를 활용한 최적화 ver
```
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.ArrayDeque;
import java.util.Deque;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws Exception {
        // 1. 빠른 입력을 위한 BufferedReader (sys.stdin.readline 역할)
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        // 2. 빠른 출력을 위한 StringBuilder (출력값들을 모아두는 창고)
        StringBuilder sb = new StringBuilder();
        
        // 3. 스택으로 사용할 Deque (ArrayDeque가 Stack 클래스보다 빠름)
        Deque<Integer> stack = new ArrayDeque<>();
        
        // 첫 번째 줄의 명령어 개수 N 읽기
        int N = Integer.parseInt(br.readLine());
        
        for (int i = 0; i < N; i++) {
            // 한 줄을 읽고 공백으로 쪼개기
            StringTokenizer st = new StringTokenizer(br.readLine());
            String command = st.nextToken();
            
            switch (command) {
                case "push":
                    // push는 뒤에 숫자가 하나 더 있으므로 한 번 더 꺼냄
                    int X = Integer.parseInt(st.nextToken());
                    stack.push(X);
                    break;
                    
                case "pop":
                    // 스택이 비어있으면 -1, 아니면 pop 결과 추가
                    if (stack.isEmpty()) {
                        sb.append("-1\n");
                    } else {
                        sb.append(stack.pop()).append("\n");
                    }
                    break;
                    
                case "size":
                    sb.append(stack.size()).append("\n");
                    break;
                    
                case "empty":
                    sb.append(stack.isEmpty() ? "1\n" : "0\n");
                    break;
                    
                case "top":
                    // 꺼내지 않고 맨 위만 확인 (peek)
                    if (stack.isEmpty()) {
                        sb.append("-1\n");
                    } else {
                        sb.append(stack.peek()).append("\n");
                    }
                    break;
            }
        }
        
        // 4. 쌓아둔 모든 결과를 한 번에 출력
        System.out.print(sb.toString());
    }
}
